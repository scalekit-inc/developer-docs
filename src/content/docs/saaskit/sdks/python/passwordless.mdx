Use `scalekit_client.passwordless` for email magic links and OTPs.

### `send_passwordless_email`

<div class="sdk-method-section">
  <CB.ClassBrowser name="Passwordless" type="client" language="python" source="https://github.com/scalekit-inc/scalekit-sdk-python/blob/main/scalekit/passwordless.py">
    <CB.ClassMethod name="send_passwordless_email" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Call `send_passwordless_email`.

      <CB.ClassMethodInput name="email" type="str">
        Email address.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="template" type="Optional[Union[str, 'TemplateType']]">
        Template.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="magiclink_auth_uri" type="Optional[str]">
        Magiclink auth uri.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="state" type="Optional[str]">
        Opaque state string.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="expires_in" type="Optional[int]">
        Expires in.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="template_variables" type="Optional[Dict[str, str]]">
        Template variables.
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="SendPasswordlessResponse">
        Send passwordless.
      </CB.ClassMethodOutput>

```python wrap showLineNumbers=false
response = scalekit_client.passwordless.send_passwordless_email(
    'user@example.com',
    template='SIGNIN',
    state='random-state',
    expires_in=3600
)

print(f'Auth Request ID: {response[0].auth_request_id}')
```

### `verify_passwordless_email`

<div class="sdk-method-section">
  <CB.ClassBrowser name="Passwordless" type="client" language="python" source="https://github.com/scalekit-inc/scalekit-sdk-python/blob/main/scalekit/passwordless.py">
    <CB.ClassMethod name="verify_passwordless_email" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Method to verify passwordless authentication

      <CB.ClassMethodInput name="code" type="Optional[str]">
        Optional. The OTP code received via email
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="link_token" type="Optional[str]">
        Optional. The token from the magic link URL
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="auth_request_id" type="Optional[str]">
        Optional. The authentication request ID from send request
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="VerifyPasswordLessResponse">
        Verify Passwordless Response
      </CB.ClassMethodOutput>

```python wrap showLineNumbers=false
response = scalekit_client.passwordless.verify_passwordless_email(
    code='123456',
    auth_request_id='auth_request_id'
)

print(f'Email: {response[0].email}')
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>
