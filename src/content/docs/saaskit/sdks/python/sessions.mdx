### `get_user_sessions`

<div class="sdk-method-section">
  <CB.ClassBrowser name="Sessions" type="client" language="python">
    <CB.ClassMethod name="get_user_sessions" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Retrieves all session details for a user with pagination and filtering support.

      <CB.ClassMethodInput name="arg1" type="unknown">
        Argument (f\Session: session.id\)
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="unknown">
        Return value of this method
      </CB.ClassMethodOutput>

```python wrap showLineNumbers=false
response = scalekit_client.sessions.get_user_sessions(
    'usr_123456',
    page_size=50
)

for session in response[0].sessions:
    print(f'Session: {session.id}')
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
