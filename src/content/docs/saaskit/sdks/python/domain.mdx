### `list_domains`

<div class="sdk-method-section">
  <CB.ClassBrowser name="Domain" type="client" language="python">
    <CB.ClassMethod name="list_domains" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Lists all domains for an organization.

      <CB.ClassMethodInput name="arg1" type="unknown">
        Argument (f"Domain: domain.domain")
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="unknown">
        Return value of this method
      </CB.ClassMethodOutput>

```python wrap showLineNumbers=false
response = scalekit_client.domain.list_domains(
    'org_123456',
    domain_type='ORGANIZATION_DOMAIN'
)

for domain in response[0].domains:
    print(f"Domain: {domain.domain}")
```
