  <CB.ClassBrowser name="Directory" type="client" language="python" source="https://github.com/scalekit-inc/scalekit-sdk-python/blob/main/scalekit/directory.py">

      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="ListDirectoriesResponse">

      </CB.ClassMethodOutput>

### create_directory
<div class="sdk-method-section">
  <CB.ClassBrowser name="Directory" type="client" language="python" source="https://github.com/scalekit-inc/scalekit-sdk-python/blob/main/scalekit/directory.py">
    <CB.ClassMethod name="create_directory" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Method to create directory based on given organization id

      <CB.ClassMethodInput name="organization_id" type="str">
        Organization id to create directory for
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="directory" type="CreateDirectory">
        CreateDirectory object with expected values for dir creation
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="CreateDirectoryResponse">
        Create Directory Response
      </CB.ClassMethodOutput>

```python wrap showLineNumbers=false
from scalekit.v1.directories.directories_pb2 import CreateDirectory

directory = CreateDirectory()
directory.provider = "azure"

response = scalekit_client.directory.create_directory(
    'org_123456',
    directory
)
```
