      <CB.ClassMethodInput name="options" type="object" hint="template, state, magiclinkAuthUri, expiresIn, templateVariables">
        Optional.
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="SendPasswordlessResponse">
        The response containing: - authRequestId: Unique identifier for the passwordless…
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
// template: The template type (SIGNIN/SIGNUP)
// state: Optional state parameter to maintain state between reques…
// magiclinkAuthUri: Optional auth URI for magic link authentication
// expiresIn: Optional expiration time in seconds (default: 3600)
// templateVariables: Optional template variables
// containing: -
// authRequestId: Unique identifier for the passwordless authentication req…
// expiresAt: Expiration time in seconds since epoch -…
const response = await scalekit.passwordless.sendPasswordlessEmail(
  'user@example.com',
  {
    template: 'SIGNIN',
    state: 'random-state',
    expiresIn: 3600
  }
);

console.log('Auth Request ID:', response.authRequestId);
```
