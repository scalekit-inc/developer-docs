Use the `passwordless` client for email magic links and OTPs.

### sendPasswordlessEmail
<div class="sdk-method-section">
  <CB.ClassBrowser name="Passwordless" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/passwordless.ts">
    <CB.ClassMethod name="sendPasswordlessEmail" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Send a passwordless authentication email

      <CB.ClassMethodInput name="email" type="string">
        The email address to send the passwordless link to
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="options" type="object" hint="template, state, magiclinkAuthUri, expiresIn, templateVariables">
        Optional settings.
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="SendPasswordlessResponse">
        Send passwordless.
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
// template: template
// state: Opaque state string
// magiclinkAuthUri: magiclink auth uri
// expiresIn: expires in
// templateVariables: template variables
const response = await scalekit.passwordless.sendPasswordlessEmail(
  'user@example.com',
  {
    template: 'SIGNIN',
    state: 'random-state',
    expiresIn: 3600
  }
);

console.log('Auth Request ID:', response.authRequestId);
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>

### verifyPasswordlessEmail
<div class="sdk-method-section">
  <CB.ClassBrowser name="Passwordless" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/passwordless.ts">
    <CB.ClassMethod name="verifyPasswordlessEmail" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Verify a passwordless authentication code or link token

      <CB.ClassMethodInput name="credential" type="object" hint="code, linkToken">
        code, linkToken.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="authRequestId" type="string">
        Optional. Optional auth request ID from the send response
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="VerifyPasswordLessResponse">
        Verified claims.
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
// code: Authorization code
// linkToken: link token
const response = await scalekit.passwordless.verifyPasswordlessEmail(
  { code: '123456' },
  'auth_request_id'
);

console.log('Email:', response.email);
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>
