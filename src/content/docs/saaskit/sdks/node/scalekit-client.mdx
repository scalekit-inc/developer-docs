### `authenticateWithCode`

<div class="sdk-method-section">
  <CB.ClassBrowser name="ScalekitClient" type="class" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/scalekit.ts">
    <CB.ClassMethod name="authenticateWithCode" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Exchanges an authorization code for access tokens and user information.

      <CB.ClassMethodInput name="code" type="string">
        The authorization code received in the callback URL after.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="redirectUri" type="string">
        The same redirect URI used in getAuthorizationUrl().
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="options" type="AuthenticationOptions">
        Optional settings.
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="AuthenticationResponse">
        Tokens and user claims.
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
app.get('/auth/callback', async (req, res) => {
  const { code } = req.query;

  const result = await scalekit.authenticateWithCode(
    code,
    'https://yourapp.com/auth/callback'
  );

  req.session.accessToken = result.accessToken;
  req.session.user = result.user;

  res.redirect('/dashboard');
});
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>

### `getIdpInitiatedLoginClaims`

<div class="sdk-method-section">
  <CB.ClassBrowser name="ScalekitClient" type="class" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/scalekit.ts">
    <CB.ClassMethod name="getIdpInitiatedLoginClaims" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Extracts and validates claims from an IdP-initiated login token.

      <CB.ClassMethodInput name="idpInitiatedLoginToken" type="string">
        The token received in the 'idp_initiated_login' query.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="options" type="TokenValidationOptions">
        Optional settings.
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="IdpInitiatedLoginClaims">
        The connection id.
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
app.get('/auth/callback', async (req, res) => {
  const { idp_initiated_login } = req.query;

  if (idp_initiated_login) {
    const claims = await scalekit.getIdpInitiatedLoginClaims(idp_initiated_login);

    const authUrl = scalekit.getAuthorizationUrl(
      'https://yourapp.com/auth/callback',
      {
        connectionId: claims.connection_id,
        organizationId: claims.organization_id,
        loginHint: claims.login_hint,
        ...(claims.relay_state && { state: claims.relay_state })
      }
    );

    return res.redirect(authUrl);
  }
});
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>
