### listPermissions
<div class="sdk-method-section">
  <CB.ClassBrowser name="Permissions" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/permission.ts">
    <CB.ClassMethod name="listPermissions" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Lists all permissions with pagination support.

      <CB.ClassMethodInput name="pageToken" type="string">
        Optional. Token for retrieving the next page
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="pageSize" type="number">
        Optional. Number of permissions per page (max: 100)
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="ListPermissionsResponse">
        Paginated list of permissions
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
const response = await scalekit.permission.listPermissions();
response.permissions.forEach(perm => {
  console.log(`${perm.name}: ${perm.description}`);
});
```

    </CB.ClassMethod>
  </CB.ClassBrowser>

### addPermissionsToRole
<div class="sdk-method-section">
  <CB.ClassBrowser name="Permissions" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/permission.ts">
    <CB.ClassMethod name="addPermissionsToRole" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Grants additional permissions to a role without removing existing assignments.

      <CB.ClassMethodInput name="roleName" type="string">
        Target role to enhance
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="permissionNames" type="string[]">
        Array of permission identifiers to add
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="AddPermissionsToRoleResponse">

      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
await scalekit.permission.addPermissionsToRole('editor', [
  'read:documents',
  'write:documents',
  'edit:documents'
]);
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>
