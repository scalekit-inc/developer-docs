Use auth helpers on the Scalekit client for login and token flows.

### `updateLoginUserDetails`

<div class="sdk-method-section">
  <CB.ClassBrowser name="Auth" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/auth.ts">
    <CB.ClassMethod name="updateLoginUserDetails" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Updates user details for an ongoing authentication request.

      <CB.ClassMethodInput name="connectionId" type="string">
        The SSO connection ID being used for authentication
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="loginRequestId" type="string">
        The unique login request identifier from the auth flow
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="user" type="UserInput">
        User details to update or associate with the login request email: User's email address sub: Unique user identifier (subject)
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="object" hint="MessageShape(typeof EmptySchema)">
        Empty response on successful update
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
await scalekit.auth.updateLoginUserDetails(
  'conn_abc123',
  'login_xyz789',
  {
    email: 'john.doe@company.com',
    sub: 'unique_user_id_456',
  }
);
```
