### `listDirectories`

<div class="sdk-method-section">
  <CB.ClassBrowser name="Directories" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/directory.ts">
    <CB.ClassMethod name="listDirectories" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Retrieves all SCIM directories configured for an organization.

      <CB.ClassMethodInput name="organizationId" type="string">
        The organization ID to list directories for (format: "org_...")
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="ListDirectoriesResponse">
        Response containing: - directories: Array of directory objects.
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
// containing: -
// directories: Array of directory objects
// id: Unique directory identifier -
// organizationId: Parent organization ID -
// provider: I…
const response = await scalekit.directory.listDirectories('org_123456');

console.log(`Found ${response.directories.length} directories`);
response.directories.forEach(dir => {
  console.log(`- ${dir.provider}: ${dir.enabled ? 'Active' : 'Inactive'}`);
  console.log(`  Users: ${dir.userCount}, Groups: ${dir.groupCount}`);
});
```

    </CB.ClassMethod>
