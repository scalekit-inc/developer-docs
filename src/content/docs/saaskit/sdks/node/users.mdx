### `createUserAndMembership`

<div class="sdk-method-section">
  <CB.ClassBrowser name="Users" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/user.ts">
    <CB.ClassMethod name="createUserAndMembership" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Creates a new user and adds them as a member of an organization in a single operation.

      <CB.ClassMethodInput name="organizationId" type="string">
        The organization ID to add the user to
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="options" type="CreateUserRequest">
        User creation configuration email: User's email address (required, must be unique) userProfile: Optional user profile information userProfile.firstName: User's first name userProfile.lastName: User's last name metadata: Custom metadata key-value pairs sendInvitationEmail: =true] - Whether to send invitation email to the user
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="CreateUserAndMembershipResponse">
        Response containing: - user: The created user object with profile and membership details
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
const response = await scalekit.user.createUserAndMembership(
  'org_123456',
  {
    email: 'john.doe@company.com',
    userProfile: {
      firstName: 'John',
      lastName: 'Doe'
    },
    sendInvitationEmail: true,
    metadata: {
      department: 'Engineering'
    }
  }
);

console.log('User created:', response.user.id);
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>
