Use the `session` client to inspect and revoke user sessions.

### getSession
<div class="sdk-method-section">
  <CB.ClassBrowser name="Sessions" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/session.ts">
    <CB.ClassMethod name="getSession" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Retrieves comprehensive metadata and status for a specific user session.

      <CB.ClassMethodInput name="sessionId" type="string">
        The session identifier to retrieve (format: "ses_...")
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="SessionDetails">
        The session id.
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
const session = await scalekit.session.getSession('ses_123456');

console.log('Status:', session.status);
console.log('User:', session.userId);
console.log('Device:', session.deviceInfo);
console.log('Location:', session.geoLocation);
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>

### getUserSessions
<div class="sdk-method-section">
  <CB.ClassBrowser name="Sessions" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/session.ts">
    <CB.ClassMethod name="getUserSessions" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Call `getUserSessions`.

      <CB.ClassMethodInput name="userId" type="string">
        User ID.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="options" type="object" hint="pageSize, pageToken, filter, status, startTime, endTime">
        Optional settings.
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="UserSessionDetails">
        The resource.
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
// pageSize: Results per page (default 10, max 100)
// pageToken: Token for the next page
// filter: Scoped tool filter
// status: status
// startTime: start time
// endTime: end time
const response = await scalekit.session.getUserSessions('usr_123456', {
  filter: { status: ['active'] },
  pageSize: 20
});

console.log(`User has ${response.totalSize} active sessions`);
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>

### revokeAllUserSessions
<div class="sdk-method-section">
  <CB.ClassBrowser name="Sessions" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/session.ts">
    <CB.ClassMethod name="revokeAllUserSessions" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

console.log(`Revoked ${response.totalCount} sessions`);
