Use the `webauthn` client to register and authenticate passkeys (WebAuthn credentials).

Build registration and authentication ceremonies with the options and verification methods below.

### listCredentials
<div class="sdk-method-section">
  <CB.ClassBrowser name="WebAuthn" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/webauthn.ts">
    <CB.ClassMethod name="listCredentials" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      List all WebAuthn credentials for a user

      <CB.ClassMethodInput name="userId" type="string">
        The user ID to list credentials for
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="ListCredentialsResponse">
        Paginated credentials.
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
const response = await scalekit.webauthn.listCredentials('usr_123456');
console.log('Credentials:', response.credentials);
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>

### updateCredential
<div class="sdk-method-section">
  <CB.ClassBrowser name="WebAuthn" type="client" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/webauthn.ts">
    <CB.ClassMethod name="updateCredential" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Update a WebAuthn credential's display name

      <CB.ClassMethodInput name="credentialId" type="string">
        The credential ID to update
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="displayName" type="string">
        The new display name for the credential
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="UpdateCredentialResponse">
        The updated resource.
      </CB.ClassMethodOutput>

```typescript wrap showLineNumbers=false
const response = await scalekit.webauthn.updateCredential(
  'cred_123',
  'My YubiKey'
);
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
