      </CB.ClassMethodInput>

      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="arg4" type="unknown">
        Argument (expiry)
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="arg5" type="string">
        e.g. "Production access token"
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="unknown">
        Return value of this method
      </CB.ClassMethodOutput>

```java wrap showLineNumbers=false
import java.util.HashMap;
import java.util.Map;
import com.google.protobuf.Timestamp;

Map<String, String> claims = new HashMap<>();
claims.put("environment", "production");

Timestamp expiry = Timestamp.newBuilder()
  .setSeconds(System.currentTimeMillis() / 1000 + 86400)
  .build();

client.tokens().create("org_123", "user_123", claims, expiry, "Production access token");
```
