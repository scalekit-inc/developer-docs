Create and validate API tokens.

### create
<div class="sdk-method-section">
  <CB.ClassBrowser name="Tokens" type="client" language="java" source="https://github.com/scalekit-inc/scalekit-sdk-java/blob/main/src/main/java/com/scalekit/api/TokenClient.java">
    <CB.ClassMethod name="create" open={true}>

      Creates a new API token for an organization with custom options.

      <CB.ClassMethodInput name="organizationId" type="String">
        The organization ID to scope the token to
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="userId" type="String">
        Optional user ID to scope the token to a specific user
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="customClaims" type="Map(String, String)">
        Optional custom claims key-value pairs
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="expiry" type="Timestamp">
        Optional expiry timestamp
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="description" type="String">
        Optional human-readable description
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="CreateTokenResponse">
        The created resource.
      </CB.ClassMethodOutput>

```java wrap showLineNumbers=false
import java.util.HashMap;
import java.util.Map;
import com.google.protobuf.Timestamp;

Map<String, String> claims = new HashMap<>();
claims.put("environment", "production");

Timestamp expiry = Timestamp.newBuilder()
  .setSeconds(System.currentTimeMillis() / 1000 + 86400)
  .build();

scalekitClient.tokens().create("org_123", "user_123", claims, expiry, "Production access token");
```

### update
<div class="sdk-method-section">
  <CB.ClassBrowser name="Tokens" type="client" language="java" source="https://github.com/scalekit-inc/scalekit-sdk-java/blob/main/src/main/java/com/scalekit/api/TokenClient.java">
    <CB.ClassMethod name="update" open={true}>

      Updates the custom claims and/or description of an existing API token.

      <CB.ClassMethodInput name="token" type="String">
        The opaque token string or token_id (apit_xxxxx)
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="customClaims" type="Map(String, String)">
        Claims to merge; set value to "" to remove a claim
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="description" type="String">
        Replacement description; null leaves unchanged, empty.
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="UpdateTokenResponse">
        The updated resource.
      </CB.ClassMethodOutput>

```java wrap showLineNumbers=false
import java.util.HashMap;
import java.util.Map;

Map<String, String> newClaims = new HashMap<>();
newClaims.put("environment", "staging");
newClaims.put("old_claim", "");  // Remove this claim

scalekitClient.tokens().update("apit_xxxxx", newClaims, "Updated description");
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
