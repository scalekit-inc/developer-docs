      </CB.ClassMethodInput>

import com.scalekit.internal.http.AuthorizationUrlOptions;
import java.net.URL;

AuthorizationUrlOptions options = new AuthorizationUrlOptions();
options.setOrganizationId("org_123");
options.setState("random-state-value");
