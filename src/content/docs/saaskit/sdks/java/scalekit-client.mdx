      <CB.ClassMethodInput name="arg1" type="string">

      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="arg2" type="string">

      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="unknown">
        Return value of this method
      </CB.ClassMethodOutput>

```java wrap showLineNumbers=false
import com.scalekit.ScalekitClient;

ScalekitClient client = new ScalekitClient(
  "https://<your-env>.scalekit.com",
  "<client_id>",
  "<client_secret>"
);
```
