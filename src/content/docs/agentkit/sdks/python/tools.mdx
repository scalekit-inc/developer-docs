### `actions.tools.list_tools`

<div class="sdk-method-section">
  <CB.ClassBrowser name="ToolsClient" type="class" language="python">
    <CB.ClassMethod name="actions.tools.list_tools" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Call `actions.tools.list_tools`.

      <CB.ClassMethodInput name="filter" type="Filter">
        Filter by provider, identifier, or tool name
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="page_size" type="int">
        Maximum tools per page.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="page_token" type="str">
        Opaque cursor from a previous list response
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="ListToolsResponse">
        List tools.
      </CB.ClassMethodOutput>


```python wrap showLineNumbers=false
from scalekit.v1.tools.tools_pb2 import Filter

response = scalekit_client.tools.list_tools(
    filter=Filter(query="calendar"),
    page_size=50,
)
```

  </CB.ClassBrowser>

### `actions.tools.list_scoped_tools`

<div class="sdk-method-section">
  <CB.ClassBrowser name="ToolsClient" type="class" language="python">
    <CB.ClassMethod name="actions.tools.list_scoped_tools" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Lists tools scoped to a specific user.

      <CB.ClassMethodInput name="identifier" type="str">
        User connected account identifier **Required.**
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="filter" type="ScopedToolFilter">
        Filter by providers, tool names, or connection names
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="page_size" type="int">
        Maximum tools per page.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="page_token" type="str">
        Opaque cursor from a previous list response
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="ListScopedToolsResponse">
        List scoped tools.
      </CB.ClassMethodOutput>


```python wrap showLineNumbers=false
from scalekit.v1.tools.tools_pb2 import ScopedToolFilter

response = scalekit_client.tools.list_scoped_tools(
    "user@example.com",
    filter=ScopedToolFilter(),
)
```

    </CB.ClassMethod>
  </CB.ClassBrowser>

### `actions.tools.execute_tool`

<div class="sdk-method-section">
  <CB.ClassBrowser name="ToolsClient" type="class" language="python">
    <CB.ClassMethod name="actions.tools.execute_tool" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Low-level tool execution.

      <CB.ClassMethodInput name="tool_name" type="str">
        Registered tool name to execute **Required.**
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="identifier" type="str">
        End-user identifier.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="params" type="dict">
        Tool arguments matching the tool input schema
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="connected_account_id" type="str">
        Connected account ID (ca_...) when you already know it
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="ExecuteToolResponse">
        ExecuteToolResponse.
      </CB.ClassMethodOutput>


```python wrap showLineNumbers=false
# Low-level ToolsClient (tool_name, identifier, params)
response = scalekit_client.tools.execute_tool(
    tool_name="gmail.messages.list",
    identifier="user@example.com",
    params={"maxResults": 10},
)
```
