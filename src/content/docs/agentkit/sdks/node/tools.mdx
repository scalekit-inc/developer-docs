`scalekit.tools` exposes raw tool schemas for custom adapters.

### `listTools`

<div class="sdk-method-section">
  <CB.ClassBrowser name="ToolsClient" type="class" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/tools.ts">
    <CB.ClassMethod name="listTools" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Lists tools available in your workspace with optional filtering and pagination.

      <CB.ClassMethodInput name="options" type="object" hint="filter, pageSize, pageToken">
        Optional settings.
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="ListToolsResponse">
        Paginated tools.
      </CB.ClassMethodOutput>


```typescript wrap showLineNumbers=false
const res = await scalekit.tools.listTools({
  pageSize: 50,
  filter: { query: 'calendar' },
});
// res.tools, res.nextPageToken
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>

### `listScopedTools`

<div class="sdk-method-section">
  <CB.ClassBrowser name="ToolsClient" type="class" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/tools.ts">
    <CB.ClassMethod name="listScopedTools" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Lists tools that are scoped to a specific connected account identifier.

      <CB.ClassMethodInput name="identifier" type="string">
        Connected account identifier to scope the tools list.
      </CB.ClassMethodInput>
      <CB.ClassMethodInput name="options" type="object" hint="filter, pageSize, pageToken">
        Optional settings.
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="ListScopedToolsResponse">
        Paginated results.
      </CB.ClassMethodOutput>


```typescript wrap showLineNumbers=false
// options.filter is required
const res = await scalekit.tools.listScopedTools('user@example.com', {
  filter: {
    // providers, toolNames, connectionNames as needed
  },
  pageSize: 50,
});
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
</div>

### `executeTool`

<div class="sdk-method-section">
  <CB.ClassBrowser name="ToolsClient" type="class" language="typescript" source="https://github.com/scalekit-inc/scalekit-sdk-node/blob/main/src/tools.ts">
    <CB.ClassMethod name="executeTool" open={true}>
      <CB.ClassModifier type="async" slot="modifiers" />

      Executes a tool using credentials from a connected account.

      <CB.ClassMethodInput name="params" type="object" hint="toolName, identifier, params, connectedAccountId, connector, organizationId, userId">
        Tool execution options.
      </CB.ClassMethodInput>
      <CB.ClassMethodOutput type="ExecuteToolResponse">
        Tool result and execution ID.
      </CB.ClassMethodOutput>


```typescript wrap showLineNumbers=false
// Low-level tools client (params, not toolInput)
await scalekit.tools.executeTool({
  toolName: 'slack.chat.postMessage',
  identifier: 'T0123456',
  params: { channel: 'C0123', text: 'hi' },
});
```

    </CB.ClassMethod>
  </CB.ClassBrowser>
