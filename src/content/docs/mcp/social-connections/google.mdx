import { getEntry, render } from 'astro:content'

export default async function Page(props) {
  const googleGuide = await getEntry('docs', 'guides/integrations/social-connections/google')
  const { Content } = await render(googleGuide)

  if (import.meta.env.DEV) {
    const fieldsToCheck = ['title', 'description', 'sidebar', 'tableOfContents', 'prev', 'next']
    const mismatches = fieldsToCheck.filter((field) => {
      const ourValue = structuredClone(frontmatter[field])
      const theirValue = structuredClone(googleGuide.data[field])
      return JSON.stringify(ourValue) !== JSON.stringify(theirValue)
    })
    if (mismatches.length) {
      console.warn(
        `mcp/auth-methods/google frontmatter differs from guides/integrations/social-connections/google: ${mismatches.join(', ')}`,
      )
    }
  }

  return (
    <>
    <Content {...props} />
    </>
  )
}
