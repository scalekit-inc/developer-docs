      // Using a self-executing function to avoid global namespace pollution
      (function() {
        // Function to initialize the copy functionality
        function initHeadingCopy() {
          const headings = document.querySelectorAll('.sl-markdown-content h2, .sl-markdown-content h3, .sl-markdown-content h4');

          headings.forEach(heading => {
            // Skip if already initialized
            if (heading.hasAttribute('data-copy-initialized')) return;
            heading.setAttribute('data-copy-initialized', 'true');

            heading.addEventListener('click', async () => {
              const headingText = heading.textContent.replace(/\s📋$/, '').trim();

              try {
                await navigator.clipboard.writeText(headingText);

                // Visual feedback
                const originalText = heading.textContent;
                heading.textContent = heading.textContent.replace(/\s📋$/, '') + ' ✓';

                setTimeout(() => {
                  heading.textContent = originalText;
                }, 2000);
              } catch (err) {
                console.error('Failed to copy heading:', err);
              }
            });
          });
        }

        // Initialize on DOMContentLoaded
        document.addEventListener('DOMContentLoaded', initHeadingCopy);

        // Also initialize now in case the DOM is already loaded
        if (document.readyState === 'complete' || document.readyState === 'interactive') {
          setTimeout(initHeadingCopy, 1);
        }

        // For Astro's client-side navigation
        document.addEventListener('astro:page-load', initHeadingCopy);
      })();

## Tool list

## `googledrive_get_file_metadata`

Retrieve metadata for a specific file in Google Drive by its file ID. Returns name, MIME type, size, creation time, and more.

| Properties | Description | Type |
| --- | --- | --- |
| `fields` | Fields to include in the response | string | null |
| `file_id` | The ID of the file to retrieve metadata for | string |
| `schema_version` | Optional schema version to use for tool execution | string | null |
| `supports_all_drives` | Support shared drives | boolean | null |
| `tool_version` | Optional tool version to use for execution | string | null |

## `googledrive_search_content`

Search inside the content of files stored in Google Drive using full-text search. Finds files where the body text matches the search term.

| Properties | Description | Type |
| --- | --- | --- |
| `fields` | Fields to include in the response | string | null |
| `mime_type` | Filter results by MIME type | string | null |
| `page_size` | Number of files to return per page | integer | null |
| `page_token` | Token for the next page of results | string | null |
| `schema_version` | Optional schema version to use for tool execution | string | null |
| `search_term` | Text to search for inside file contents | string |
| `supports_all_drives` | Include shared drives in results | boolean | null |
| `tool_version` | Optional tool version to use for execution | string | null |

## `googledrive_search_files`

Search for files and folders in Google Drive using query filters like name, type, owner, and parent folder.

| Properties | Description | Type |
| --- | --- | --- |
| `fields` | Fields to include in the response | string | null |
| `order_by` | Sort order for results | string | null |
| `page_size` | Number of files to return per page | integer | null |
| `page_token` | Token for the next page of results | string | null |
| `query` | Drive search query string | string | null |
| `schema_version` | Optional schema version to use for tool execution | string | null |
| `supports_all_drives` | Include shared drives in results | boolean | null |
| `tool_version` | Optional tool version to use for execution | string | null |
