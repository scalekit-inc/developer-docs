      // Using a self-executing function to avoid global namespace pollution
      (function() {
        // Function to initialize the copy functionality
        function initHeadingCopy() {
          const headings = document.querySelectorAll('.sl-markdown-content h2, .sl-markdown-content h3, .sl-markdown-content h4');

          headings.forEach(heading => {
            // Skip if already initialized
            if (heading.hasAttribute('data-copy-initialized')) return;
            heading.setAttribute('data-copy-initialized', 'true');

            heading.addEventListener('click', async () => {
              const headingText = heading.textContent.replace(/\s📋$/, '').trim();

              try {
                await navigator.clipboard.writeText(headingText);

                // Visual feedback
                const originalText = heading.textContent;
                heading.textContent = heading.textContent.replace(/\s📋$/, '') + ' ✓';

                setTimeout(() => {
                  heading.textContent = originalText;
                }, 2000);
              } catch (err) {
                console.error('Failed to copy heading:', err);
              }
            });
          });
        }

        // Initialize on DOMContentLoaded
        document.addEventListener('DOMContentLoaded', initHeadingCopy);

        // Also initialize now in case the DOM is already loaded
        if (document.readyState === 'complete' || document.readyState === 'interactive') {
          setTimeout(initHeadingCopy, 1);
        }

        // For Astro's client-side navigation
        document.addEventListener('astro:page-load', initHeadingCopy);
      })();

## Tool list

## `outlook_create_calendar_event`

Create a new calendar event in the user's Outlook calendar. Supports attendees, recurrence, reminders, online meetings, multiple locations, and event properties.

| Properties | Description | Type |
| --- | --- | --- |
| `attendees_optional` | Array of email addresses for optional attendees | string | null |
| `attendees_required` | Array of email addresses for required attendees | string | null |
| `attendees_resource` | Array of email addresses for resources (meeting rooms, equipment) | string | null |
| `body_content` | No description | string | null |
| `body_contentType` | No description | string | null |
| `end_datetime` | No description | string |
| `end_timezone` | No description | string |
| `hideAttendees` | When true, each attendee only sees themselves | boolean | null |
| `importance` | Event importance level | string | null |
| `isAllDay` | Mark as all-day event | boolean | null |
| `isOnlineMeeting` | Create an online meeting (Teams/Skype) | boolean | null |
| `isReminderOn` | Enable or disable reminder | boolean | null |
| `location` | No description | string | null |
| `locations` | JSON array of location objects with displayName, address, coordinates | string | null |
| `onlineMeetingProvider` | Online meeting provider | string | null |
| `recurrence_days_of_week` | Days of week for weekly recurrence (comma-separated) | string | null |
| `recurrence_end_date` | End date for recurrence (YYYY-MM-DD), required if range_type is endDate | string | null |
| `recurrence_interval` | How often the event recurs (e.g., every 2 weeks = 2) | integer | null |
| `recurrence_occurrences` | Number of occurrences, required if range_type is numbered | integer | null |
| `recurrence_range_type` | How the recurrence ends | string | null |
| `recurrence_start_date` | Start date for recurrence (YYYY-MM-DD) | string | null |
| `recurrence_type` | Recurrence pattern type | string | null |
| `reminderMinutesBeforeStart` | Minutes before event start to show reminder | integer | null |
| `sensitivity` | Event sensitivity/privacy level | string | null |
| `showAs` | Free/busy status | string | null |
| `start_datetime` | No description | string |
| `start_timezone` | No description | string |
| `subject` | No description | string |

| Properties | Description | Type |

## `outlook_list_calendar_events`

List calendar events from the user's Outlook calendar with filtering, sorting, pagination, and field selection.

| Properties | Description | Type |
| --- | --- | --- |
| `filter` | OData filter expression to filter events (e.g., startsWith(subject,'All')) | string | null |
| `orderby` | OData orderby expression to sort events (e.g., start/dateTime desc) | string | null |
| `select` | Comma-separated list of properties to include in the response | string | null |
| `skip` | Number of events to skip for pagination | number | null |
| `top` | Maximum number of events to return | number | null |
