Once a connected account is set up, make API calls through the Scalekit proxy. Scalekit injects the Diarize API key automatically — you never handle credentials in your application code.

You can interact with Diarize in two ways — via direct proxy API calls or via Scalekit optimized tool calls.

## Proxy API calls

A typical Diarize workflow has three steps: submit a job, poll until complete, then download the transcript.

<Tabs syncKey="tech-stack">
  <TabItem label="Node.js">
    ```typescript
    import { ScalekitClient } from '@scalekit-sdk/node';
    import 'dotenv/config';

    const connectionName = 'diarize'; // connection name from your Scalekit dashboard
    const identifier = 'user_123';    // your user's unique identifier

    // Get your credentials from app.scalekit.com → Developers → Settings → API Credentials
    const scalekit = new ScalekitClient(
      process.env.SCALEKIT_ENV_URL,
      process.env.SCALEKIT_CLIENT_ID,
      process.env.SCALEKIT_CLIENT_SECRET
    );
    const actions = scalekit.actions;

    // Step 1: Submit a transcription job
    const job = await actions.request({
      connectionName,
      identifier,
      path: '/api/v1/jobs',
      method: 'POST',
      body: {
        sourceUrl: 'https://www.youtube.com/watch?v=example',
        language: 'en',   // optional — omit for auto-detection
        num_speakers: 2,  // optional — helps diarization accuracy
      },
    });
    const jobId = job.data.id;
    console.log('Job created:', jobId);

    // Step 2: Poll until complete (use estimatedTime from response to set timeout)
    let status = 'pending';
    while (status !== 'completed' && status !== 'failed') {
      await new Promise(r => setTimeout(r, 10_000)); // wait 10 s between polls
      const statusRes = await actions.request({
        connectionName,
        identifier,
        path: `/api/v1/jobs/${jobId}`,
        method: 'GET',
      });
      status = statusRes.data.status;
      console.log('Status:', status);
    }

    if (status === 'failed') throw new Error('Transcription job failed');

    // Step 3: Download the transcript (json | txt | srt | vtt)
    const transcript = await actions.request({
      connectionName,
      identifier,
      path: `/api/v1/jobs/${jobId}/transcript/json`,
      method: 'GET',
    });
    console.log(transcript.data);
    ```

<Aside type="tip" title="No OAuth flow needed">
  Diarize uses API key auth — unlike OAuth connectors, there is no authorization link or redirect flow. Once you call `upsert_connected_account` (Python) / `upsertConnectedAccount` (Node.js), or add an account via the dashboard, your users can make requests immediately.
</Aside>

## Scalekit tools

Use `execute_tool` to call Diarize tools directly. Scalekit resolves the connected account, injects the API key, and returns a structured response.

### Submit a transcription job

Submit a YouTube, X, Instagram, or TikTok URL for transcription. Returns a `job_id` to use in subsequent status and download calls.

```python title="examples/diarize_create_job.py"
import scalekit.client, os
from dotenv import load_dotenv
load_dotenv()

scalekit_client = scalekit.client.ScalekitClient(
    client_id=os.getenv("SCALEKIT_CLIENT_ID"),
    client_secret=os.getenv("SCALEKIT_CLIENT_SECRET"),
    env_url=os.getenv("SCALEKIT_ENV_URL"),
)
actions = scalekit_client.actions

# Resolve connected account
response = actions.get_or_create_connected_account(
    connection_name="diarize",
    identifier="user_123"
)
connected_account = response.connected_account

# Submit transcription job for a YouTube video
result = actions.execute_tool(
    tool_name="diarize_create_transcription_job",
    connected_account_id=connected_account.id,
    tool_input={
        "url": "https://www.youtube.com/watch?v=example",
        "language": "en",   # optional — omit for auto-detection
        "num_speakers": 2,  # optional — improves speaker diarization
    }
)

job_id = result.result["id"]
print("Job submitted:", job_id)
print("Estimated processing time:", result.result.get("estimatedTime"), "seconds")
```

### Check job status

Poll this tool until `status` is `completed` or `failed`. Use the `estimatedTime` value (in seconds) returned from the create-job response to pace your polling — do not give up before that time has elapsed.

```python title="examples/diarize_check_status.py"
import time

result = actions.execute_tool(
    tool_name="diarize_get_job_status",
    connected_account_id=connected_account.id,
    tool_input={"job_id": job_id}
)

status = result.result["status"]
print("Status:", status)  # pending | processing | completed | failed
```

<Aside type="note" title="Polling guidance">
  The `estimatedTime` field (in seconds) tells you how long processing is expected to take. For a 49-minute episode, `estimatedTime` may be around 891 seconds (~15 minutes). Wait at least that long before treating the job as timed out.
</Aside>

### Download the transcript

Download a completed transcript in your preferred format. `json` includes full speaker diarization, segment timestamps, and word-level data. `txt`, `srt`, and `vtt` are plain-text formats for display or subtitle use.

```python title="examples/diarize_download_transcript.py"
# Download as structured JSON (includes speaker labels and timestamps)
result = actions.execute_tool(
    tool_name="diarize_download_transcript",
    connected_account_id=connected_account.id,
    tool_input={
        "job_id": job_id,
        "format": "json",   # json | txt | srt | vtt (default: json)
    }
)

transcript = result.result
for segment in transcript.get("segments", []):
    speaker = segment.get("speaker", "Unknown")
    text = segment.get("text", "")
    start = segment.get("start", 0)
    print(f"[{start:.1f}s] {speaker}: {text}")
```

### End-to-end workflow

Complete example: submit a job, wait for completion, and download the diarized transcript.

```python title="examples/diarize_full_workflow.py"
import scalekit.client, os, time
from dotenv import load_dotenv
load_dotenv()

scalekit_client = scalekit.client.ScalekitClient(
    client_id=os.getenv("SCALEKIT_CLIENT_ID"),
    client_secret=os.getenv("SCALEKIT_CLIENT_SECRET"),
    env_url=os.getenv("SCALEKIT_ENV_URL"),
)
actions = scalekit_client.actions

response = actions.get_or_create_connected_account(
    connection_name="diarize",
    identifier="user_123"
)
connected_account = response.connected_account

# Submit job
create_result = actions.execute_tool(
    tool_name="diarize_create_transcription_job",
    connected_account_id=connected_account.id,
    tool_input={"url": "https://www.youtube.com/watch?v=example", "language": "en"}
)
job_id = create_result.result["id"]
estimated_seconds = create_result.result.get("estimatedTime", 120)
print(f"Job {job_id} submitted. Estimated time: {estimated_seconds}s")

# Poll until complete
poll_interval = 15  # seconds
elapsed = 0
while True:
    time.sleep(poll_interval)
    elapsed += poll_interval
    status_result = actions.execute_tool(
        tool_name="diarize_get_job_status",
        connected_account_id=connected_account.id,
        tool_input={"job_id": job_id}
    )
    status = status_result.result["status"]
    print(f"[{elapsed}s] Status: {status}")

    if status == "completed":
        break
    if status == "failed":
        raise RuntimeError(f"Job {job_id} failed")
    if elapsed > estimated_seconds * 2:
        raise TimeoutError(f"Job {job_id} exceeded {elapsed}s")

# Download transcript
transcript_result = actions.execute_tool(
    tool_name="diarize_download_transcript",
    connected_account_id=connected_account.id,
    tool_input={"job_id": job_id, "format": "json"}
)

for segment in transcript_result.result.get("segments", []):
    speaker = segment.get("speaker", "?")
    text = segment.get("text", "")
    start = segment.get("start", 0)
    print(f"[{start:.1f}s] {speaker}: {text}")
```
