    <CB.ClassMethodInput name="organization_id" type="str">
        Organization tenant ID.
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="user_id" type="str">

    <CB.ClassMethodInput name="api_config" type="dict">

      </CB.ClassMethodInput>
    <CB.ClassMethodOutput type="CreateConnectedAccountResponse">

  <CB.ClassMethod name="get_connected_account">
    Fetches auth details for a connected account. Returns sensitive credentials. Protect access to this method.
    
    Use this when you know the connected account already exists and you need its credential payload. For first-time setup or general application flows, prefer `get_or_create_connected_account` so new users do not hit a not-found error.
    
    Requires `connected_account_id` **or** `connection_name` + `identifier`.

    <CB.ClassMethodInput name="connection_name" type="str">
        Connector slug.
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="identifier" type="str">
        End-user or workspace identifier. Use with connection_name.
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="connected_account_id" type="str">
        Connected account ID (ca_...) when resolving by ID instead.
      </CB.ClassMethodInput>
    <CB.ClassMethodOutput type="GetConnectedAccountAuthResponse">
        The connected account.
      </CB.ClassMethodOutput>
  </CB.ClassMethod>

  <CB.ClassMethod name="execute_tool">
    Executes a named tool via Scalekit. Pre- and post-modifiers run automatically if registered.

    <CB.ClassMethodInput name="tool_name" type="str">
        Tool name (e.g. gmail_fetch_emails) **Required.**
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="tool_input" type="dict">
        Parameters the tool expects **Required.**
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="identifier" type="str">
        User's identifier
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="connected_account_id" type="str">
        Direct connected account ID
      </CB.ClassMethodInput>
    <CB.ClassMethodOutput type="ExecuteToolResponse">
        Tool result and execution ID.
      </CB.ClassMethodOutput>
  </CB.ClassMethod>

```python title="Example"
result = actions.execute_tool(
    tool_name="gmail_fetch_emails",
    tool_input={"max_results": 5, "label": "UNREAD"},
    identifier="user@example.com",
)
emails = result.data
```

  <CB.ClassMethod name="request">
    Makes a REST API call on behalf of a connected account. Scalekit injects the user's OAuth token automatically.

    <CB.ClassMethodInput name="connection_name" type="str">
        Connector slug **Required.**
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="identifier" type="str">
        User's identifier **Required.**
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="path" type="str">
        API path (e.g. /gmail/v1/users/me/messages) **Required.**
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="method" type="str">
        HTTP method. Default: GET
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="query_params" type="dict">
        URL query parameters appended to path
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="body" type="any">
        JSON-serializable body for POST, PUT, PATCH, or similar.
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="form_data" type="dict">
        Multipart form fields when the upstream API expects form.
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="headers" type="dict">
        Extra HTTP headers merged with Scalekit-injected auth.
      </CB.ClassMethodInput>
    <CB.ClassMethodOutput type="requests.Response">
        See return type `requests.Response`.
      </CB.ClassMethodOutput>
  </CB.ClassMethod>

```python title="Example"
response = actions.request(
    connection_name="gmail",
    identifier="user@example.com",
    path="/gmail/v1/users/me/messages",
    query_params={"maxResults": 5, "q": "is:unread"},
)
messages = response.json()["messages"]
```
