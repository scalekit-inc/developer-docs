`scalekit.tools` gives access to raw tool schemas. Use this when building a custom framework adapter or passing schemas directly to an LLM API.

  name="ToolsClient"

  language="typescript"

  <CB.ClassMethod name="listTools">
    Lists all tools available in your workspace.

    <CB.ClassMethodInput name="filter" type="Filter">
        Filter by provider, identifier, or tool name
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="pageSize" type="number">
        Maximum tools per page (server default if omitted)
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="pageToken" type="string">
        Opaque cursor from a previous list response
      </CB.ClassMethodInput>
    <CB.ClassMethodOutput type="ListToolsResponse">
        Paginated tools.
      </CB.ClassMethodOutput>
  </CB.ClassMethod>

  <CB.ClassMethod name="listScopedTools">
    Lists tools scoped to a specific user. Use this for tool discovery because it returns pagination metadata such as `nextPageToken` and `totalSize`.

    <CB.ClassMethodInput name="identifier" type="string">
        User's connected account identifier **Required.**
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="filter" type="ScopedToolFilter">
        Filter by providers, tool names, or connection names
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="pageSize" type="number">
        Maximum tools per page.
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="pageToken" type="string">
        Opaque cursor from a previous list response
      </CB.ClassMethodInput>
    <CB.ClassMethodOutput type="ListScopedToolsResponse">
        Paginated tools.
      </CB.ClassMethodOutput>
  </CB.ClassMethod>

```ts title="Example"
const { tools } = await scalekit.tools.listScopedTools('user@example.com', {
  filter: { connectionNames: ['gmail'] },
  pageSize: 100,
});
// Pass tools to your LLM's tool call API
```
