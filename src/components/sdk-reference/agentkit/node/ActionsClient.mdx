  <CB.ClassMethod name="getAuthorizationLink">
    Generates a time-limited OAuth magic link to authorize a user's connection.

    <CB.ClassMethodInput name="connectionName" type="string">
        Connector slug (e.g. gmail)
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="identifier" type="string">
        User's identifier (e.g. email)
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="connectedAccountId" type="string">
        Direct connected account ID (ca_...)
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="organizationId" type="string">
        Organization tenant ID.
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="userId" type="string">
        Your app user ID.
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="state" type="string">
        Opaque value passed through to the redirect URL
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="userVerifyUrl" type="string">
        Your app's redirect URL for user verification
      </CB.ClassMethodInput>
    <CB.ClassMethodOutput type="GetMagicLinkForConnectedAccountResponse">
        Authorization magic link.
      </CB.ClassMethodOutput>
  </CB.ClassMethod>

```ts title="Example"
const { link } = await scalekit.actions.getAuthorizationLink({
  connectionName: 'gmail',
  identifier: 'user@example.com',
  userVerifyUrl: 'https://your-app.com/verify',
});
// Redirect the user to link
```

    <CB.ClassMethodInput name="provider" type="string">

    <CB.ClassMethodOutput type="ListAppConnectionsResult">
        Paginated connections.
      </CB.ClassMethodOutput>
  </CB.ClassMethod>

```ts title="list-connections.ts"
try {
  const { connections } = await scalekit.actions.listConnections({
    pageSize: 30,
    provider: 'GMAIL',
  });
  for (const conn of connections) {
    console.log(conn.id, conn.connectionName, conn.enabled);
  }
} catch (error) {
  console.error('Failed to list connections', error);
}
```

      </CB.ClassMethodInput>
    <CB.ClassMethodOutput type="ExecuteToolResponse">

```ts title="Example"

  toolName: 'gmail_fetch_emails',
  toolInput: { maxResults: 5, label: 'UNREAD' },

});
const emails = result.data;
```

  <CB.ClassMethod name="request">
    Makes a REST API call on behalf of a connected account. Scalekit injects the user's OAuth token automatically.

    <CB.ClassMethodInput name="connectionName" type="string">
        Connector slug **Required.**
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="identifier" type="string">
        User's identifier **Required.**
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="path" type="string">
        API path (e.g. /gmail/v1/users/me/messages) **Required.**
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="method" type="string">
        HTTP method. Default: GET
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="queryParams" type={"Record<string, unknown>
        "}>
      URL query parameters appended to path
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="body" type="unknown">
        JSON-serializable body for POST, PUT, PATCH, or similar.
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="formData" type={"Record<string, unknown>
        "}>
      Multipart form fields when the upstream API.
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="headers" type={"Record<string, string>
        "}>
      Extra HTTP headers merged with Scalekit-injected.
      </CB.ClassMethodInput>
    <CB.ClassMethodInput name="timeoutMs" type="number">
        Default: 30000
      </CB.ClassMethodInput>
    <CB.ClassMethodOutput type="AxiosResponse">
        See return type `AxiosResponse`.
      </CB.ClassMethodOutput>
  </CB.ClassMethod>

```ts title="Example"
const response = await scalekit.actions.request({
  connectionName: 'gmail',
  identifier: 'user@example.com',
  path: '/gmail/v1/users/me/messages',
  queryParams: { maxResults: 5, q: 'is:unread' },
});
const messages = response.data.messages;
```
